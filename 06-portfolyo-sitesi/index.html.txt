            // Ana sayfaya dön butonu varsa güncelle
            const backBtn = document.querySelector('.back-to-home');

        async function fetchWeather() {
            try {
                const res = await fetch('https://api.open-meteo.com/v1/forecast?latitude=41.9442&longitude=34.5861&current_weather=true');
                const data = await res.json();
                const temp = data.current_weather.temperature;
                const code = data.current_weather.weathercode;
                
                const widget = document.createElement('div');
                widget.className = 'weather-widget';
                widget.innerHTML = `Ayancık: ${temp}°C`;
                document.body.appendChild(widget);

                const container = document.getElementById('weather-container');
                if ([71, 73, 75, 85, 86].includes(code)) {
                    for(let i=0; i<30; i++) {
                        let s = document.createElement('div'); s.className='snowflake'; s.textContent='❄';
                        s.style.left = Math.random()*100+'%'; s.style.fontSize = Math.random()*20+10+'px';
                        s.style.animationDuration = Math.random()*3+2+'s'; container.appendChild(s);

        function sendMessage() {
            const input = document.getElementById('chat-input');
            const msg = input.value.trim();
            if(!msg) return;
            addMsg(msg, 'user');
            input.value = '';
            
            const thinkingTime = Math.floor(Math.random() * 800) + 600;

            setTimeout(() => {
                const response = getBotResponse(msg.toLocaleLowerCase('tr-TR')); // tr-TR lowercase helps even for English mostly
                const botText = typeof response === 'object' ? response.text : response;
                const botAction = typeof response === 'object' ? response.action : null;
                addMsg(botText, 'bot');
                if (botAction) botAction();
            }, thinkingTime);
        }

        function addMsg(text, sender) {
            const b = document.getElementById('chat-body');
            const d = document.createElement('div');
            d.className = `message ${sender}`;
            d.innerHTML = text;
            b.appendChild(d);
            b.scrollTop = b.scrollHeight;
        }
